let customers = [];
let currentIndex = 0;

document.getElementById('fileInput').addEventListener('change', handleFile);

function handleFile(e) {
const file = e.target.files[0];
if (!file) return;

if (file.name.endsWith('.csv')) {
Papa.parse(file, {
header: true,
complete: res => processData(res.data)
});
} else {
const reader = new FileReader();
reader.onload = (evt) => {
const data = new Uint8Array(evt.target.result);
const workbook = XLSX.read(data, { type: 'array' });
const sheet = workbook.Sheets[workbook.SheetNames[0]];
const json = XLSX.utils.sheet_to_json(sheet);
processData(json);
};
reader.readAsArrayBuffer(file);
}
}

function processData(data) {
const seen = new Set();
let counter = 1;

customers = data.map(row => {
let name = row.Name || row.name || '';
let phone = normalizePhone(row.Phone || row.phone);

if (!phone || seen.has(phone)) return null;

seen.add(phone);

if (!name) {
name = "Customer " + counter++;
}

return { name, phone };
}).filter(x => x);

render();
}

function normalizePhone(phone) {
return String(phone).replace(/\D/g, '');
}

function render() {
document.getElementById('messageSection').style.display = 'block';
document.getElementById('listSection').style.display = 'block';

const tbody = document.getElementById('tableBody');
tbody.innerHTML = '';

customers.forEach((c, i) => {
const row = document.createElement('tr');

row.innerHTML = `
<td>${i+1}</td>
<td>${c.name}</td>
<td>${c.phone}</td>
<td>
<button onclick="send(${i})">Send</button>
<button onclick="copyMsg(${i})">Copy</button>
</td>
`;

tbody.appendChild(row);
});
}

function buildMsg(c, i) {
let msg = document.getElementById('messageInput').value || "Hello {name}";
return msg
.replaceAll("{name}", c.name)
.replaceAll("{phone}", c.phone)
.replaceAll("{index}", i+1);
}

function send(i) {
const c = customers[i];
const msg = encodeURIComponent(buildMsg(c, i));
window.open(`https://wa.me/${c.phone}?text=${msg}`, '_blank');
}

function copyMsg(i) {
const text = buildMsg(customers[i], i);
navigator.clipboard.writeText(text);
alert("Copied");
}

function startSending() {
currentIndex = 0;
next();
}

function next() {
if (currentIndex >= customers.length) {
alert("Done");
return;
}

send(currentIndex);
currentIndex++;

setTimeout(next, 3000);
}
